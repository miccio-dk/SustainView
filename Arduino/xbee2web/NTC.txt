{

	  Serial.begin(9600);

void loop()
{
float temp;
 float val = analogRead(A0);
 temp = log(((10230000/val) - 10000)); 
 temp = 1 / (0.001129148 + (0.000234125 + (0.0000000876741 * temp * temp ))* temp);
 temp = temp - 273.15;
 delay(3000);

  Serial.println("Temperature");
 Serial.println(temp);
	 

}
